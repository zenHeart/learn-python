# 语句

# 条件判断

# if

利用形如

if 表达式:
  语句块

申明 if 语句示例判断输入是否为偶数

a = int(input('enter a number: '))

if a % 2 == 0 :
  print('a is even number')

# else

采用 else 表示额外的分支逻辑

a = int(input('enter a number: '))

if a % 2 == 0 :
  print('a is even number')
else:
  print('a is odd number')

# elif

采用 elif 实现多分支判断

a = int(input("enter a number: "))

if a < 0:
    print('negative number')
elif a == 0:
    print('zero')
else:
    print("positive")

# 循环

# for

利用 for 循环 list 结构

for i in [1,2,3]:
  print(i)

# while

可以采用 while 实现逻辑判断循环。 采用 while 循环 list

a = [1,2,3]
i = 0
while i < len(a):
    print(a[i])
    i+=1

# break

可以利用 break 实现跳出循环

l = [1,2,3,4]
i = 0

while i < len(l):
  if(l[i] > 2):
    print("first element greater 2 index: ", i)
    # jump to loop
    break;
  i+=1

# continue

break 会跳出整个 for 或 while 语句，若只期望 某次循环的后续执行内容，可使用 continue。

for num in [1,2,3,4]:
    if num % 2 == 0:
      print("even number: ", num)
      continue
      print("this code never execute")
    print("odd number: ", num)

# 错误捕获

# try

利用 try 捕获错误，except 中处理错误逻辑

try:
    x = int(input("Please enter a number: "))
    print('x is illegal number')
except:
    print("Oops!  That was no valid number.")

# exception

# finally

try:
    x = int(input("Please enter a number: "))
    print('x is illegal number')
except:
    print("Oops!  That was no valid number.")
finally:
    print("over")

# 知识点

  • if 语句结构

    if_stmt ::=  "if" assignment_expression ":" suite
           ("elif" assignment_expression ":" suite)*
           ["else" ":" suite]

  • for 语句

    for_stmt ::=  "for" target_list "in" expression_list ":" suite
              ["else" ":" suite]

  • try 语句

    try_stmt  ::=  try1_stmt | try2_stmt
try1_stmt ::=  "try" ":" suite
              ("except" [expression ["as" identifier]] ":" suite)+
              ["else" ":" suite]
              ["finally" ":" suite]
try2_stmt ::=  "try" ":" suite
              "finally" ":" suite

  • range(n) 定义一个循环迭代

    • range(x, y) 定义一个范围的迭代
    • range(start,end,step) 定义步长迭代

  • else 只在 for 循环耗尽或 while 判断为空时执行，而非 break 状态。素数判断逻辑

  • pass 表示空语句

    • 完成一个语句的正常结构
    • 空类
    • 函数占位，表示代办
